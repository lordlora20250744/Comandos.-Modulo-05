# =========================
# PRÁCTICA 1 - Sincronización de carpetas mediante rsync
# =========================

sudo mkdir /home/practica_rsync
# Crea la carpeta donde estarán los archivos a sincronizar

sudo chown sistemas3:sistemas3 /home/practica_rsync
# Asigna permisos al usuario sistemas3

cd /home/practica_rsync
# Entrar a la carpeta

touch laboratorio{1..100}.txt
# Crear 100 archivos de prueba

ssh sistemas3@192.168.1.100
# Probar conexión SSH al otro nodo usando la IP flotante

exit
# Salir de la conexión

rsync -av /home/practica_rsync/ sistemas3@192.168.1.100:/home/sistemas3/practica_rsync/
# Sincroniza la carpeta local al nodo remoto

nano /home/sync_rsync.sh
# Crear script automático de sincronización

#!/bin/bash
rsync -avz /home/practica_rsync/ sistemas3@192.168.1.100:/home/sistemas3/practica_rsync/

chmod +x /home/sync_rsync.sh
# Dar permisos de ejecución

ls -l /home/sync_rsync.sh
# Verificar permisos

ssh-keygen
# Generar clave SSH

ssh-copy-id sistemas3@192.168.1.100
# Copiar clave pública al otro nodo

touch /home/practica_rsync/prueba_cron.txt
# Crear archivo de prueba

/home/sync_rsync.sh
# Ejecutar sincronización manual

ssh sistemas3@192.168.1.100
# Entrar al nodo remoto

ls /home/sistemas3/practica_rsync
# Verificar archivos sincronizados

exit
# Salir

crontab -e
# Crear tarea automática

* * * * * /home/sync_rsync.sh

crontab -l
# Verificar cron configurado

touch /home/practica_rsync/cron_ok.txt
# Crear nuevo archivo para validar cron

ssh sistemas3@192.168.1.100
# Entrar nuevamente al nodo remoto

ls /home/sistemas3/practica_rsync
# Verificar que cron sincronizó

exit
# Salir


# =========================
# PRACTICA 2 – Instalación y configuración de cluster de alta disponibilidad (pacemaker)
# =========================

sudo apt update
# Actualiza paquetes

sudo apt install pacemaker corosync pcs -y
# Instala herramientas del cluster HA

sudo systemctl enable pcsd
# Habilita servicio pcsd

sudo systemctl start pcsd
# Inicia pcsd

sudo passwd hacluster
# Define contraseña del usuario hacluster

sudo pcs host auth 192.168.1.50 192.168.1.51 -u hacluster
# Autentica ambos nodos

sudo pcs cluster setup --name mi_cluster nodo1 nodo2
# Crea el cluster

sudo pcs cluster start --all
# Inicia todos los nodos

sudo pcs cluster enable --all
# Habilita inicio automático

sudo pcs property set stonith-enabled=false
# Desactiva fencing para laboratorio

sudo pcs property set no-quorum-policy=ignore
# Ignora quorum en práctica de 2 nodos

sudo pcs status
# Verifica estado

sudo pcs resource create IP_Flotante ocf:heartbeat:IPaddr2 ip=192.168.1.100 cidr_netmask=24 op monitor interval=30s
# Crea la IP flotante .100

sudo pcs status
# Confirmar recurso

ip a
# Verificar IP flotante en interfaz

ping 192.168.1.100
# Probar conectividad

sudo reboot
# Reiniciar nodo para prueba

sudo pcs status
# Confirmar que recurso sigue activo

ip a
# Revisar si la IP flotante sigue asignada
