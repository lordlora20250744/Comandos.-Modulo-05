sudo mkdir /home/practica_rsync

PRACTICA 2 – Instalación y configuración de cluster de alta disponibilidad (pacemaker)

sudo apt update
sudo apt install pacemaker corosync pcs -y

sudo systemctl enable pcsd
sudo systemctl start pcsd

sudo passwd hacluster

sudo pcs host auth 192.168.1.50 192.168.1.51 -u hacluster

sudo pcs cluster auth nodo1 nodo2 -u hacluster -p tu_password

sudo pcs cluster setup cluster_ha 192.168.1.50 192.168.1.51

sudo pcs cluster setup --name mi_cluster nodo1 nodo2

sudo pcs cluster start --all

sudo pcs cluster enable --all

sudo pcs property set stonith-enabled=false
sudo pcs property set no-quorum-policy=ignore

sudo pcs status
sudo pcs cluster status

sudo pcs resource create IP_Flotante ocf:heartbeat:IPaddr2 ip=192.168.1.100 cidr_netmask=24 op monitor interval=30s

sudo pcs status

ip a

ping 192.168.1.100

sudo reboot

sudo pcs status
ip a

PRÁCTICA 3 - Cluster de Alta Disponibilidad HTTP

sudo nano /var/www/html/index.html

sudo nano /etc/keepalived/keepalived.conf
